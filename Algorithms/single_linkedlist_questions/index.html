Node* Reverse(Node *head)
{
    if (head == NULL) {
        return NULL;
    } else {
        Node *reverseHead = NULL;
        Node *cur = head;
        Node *pre = NULL;
        Node *next = NULL;
        while(cur != NULL) {
            next = cur->next;
            if (next == NULL) {
                reverseHead = cur;
            }
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return reverseHead;
    }
}

Node* RemoveDuplicates(Node *head)
{
    if(head == NULL) {
        return NULL;
    } else {
        Node *cur = head, *fol = cur->next;
        while(fol != NULL) {
            while(fol && (fol->data == cur->data)) {
                fol = fol->next;
            }
            cur->next = fol;
            if(fol) {
                cur = fol;
                fol = cur->next;
            }
        }
        return head;
    }

bool has_cycle(Node* head) {
    if(head == NULL) {
        return false;
    } else {
        Node *p = head, *q = head;

            q = q->next->next;
            p = p->next;
            if(p == q) {
                return true;
            }
        }
        return false;
    }

            while(ptr && (data > ptr->data)) {
                cur = ptr;
                ptr = ptr->next;
            }
            if(! ptr) {
                cur->next = newNode;
                newNode->next = NULL;
                newNode->prev = cur;
                return head;
            } else {
                cur->next = newNode;
                newNode->next = ptr;
                newNode->prev = cur;
                ptr->prev = newNode;
            }
            return head;
        }
    }
}

Node* Reverse(Node* head)
{  
    if (head == NULL) {
        return NULL;
    } 
    Node *cur = head, *next = NULL,*reversedHead = NULL;
    while(cur) {
        next = cur->next;
        if (next == NULL) {
            reversedHead = cur;
        }
        cur->next = cur->prev;
        cur->prev = next;
        cur = next;
    }
    return reversedHead;
}
