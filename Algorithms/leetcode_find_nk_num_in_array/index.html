class Solution {
public:
	void printHalfNumber(vector<int> &arr) {
    	int cand = 0;
    	int times = 0;
    	for(int i = 0; i < arr.size();i++) {
        	if (times == 0) {
            	cand = arr[i];
            	times = 1;
        	} else if (arr[i] == cand) {
             	times++;
        	} else {
            	times--;
        	}
    	}
    	times = 0;
    	for(int i = 0; i < arr.size(); i++) {
        	if (arr[i] == cand) {
            	times++;
        	}
    	}
    	if (times > arr.size()/2) {
        	printf("%d\n", cand);
    	} else {
        	printf("No such num\n");
    	}
	}
};
